## Deploy por ZIP

Envie o código diretamente pelo SDK sem usar o CLI:

```typescript
import { createReadStream, statSync } from 'fs'

// Criar ZIP do projeto primeiro (use archiver, jszip, etc.)
const zipPath = './dist/meu-app.zip'

const deployment = await nivo.apps.deploy('app_id', {
  source: createReadStream(zipPath),
  size: statSync(zipPath).size
})

console.log(`Deployment ID: ${deployment.id}`)
console.log(`Status: ${deployment.status}`)
```

## Acompanhar o deploy

```typescript
// Polling simples
async function aguardarDeploy(deploymentId: string) {
  while (true) {
    const dep = await nivo.apps.getDeployment('app_id', deploymentId)

    console.log(`Status: ${dep.status}`)

    if (dep.status === 'success') {
      console.log('✔ Deploy concluído!')
      return dep
    }

    if (dep.status === 'error') {
      throw new Error(`Deploy falhou: ${dep.error}`)
    }

    await new Promise(r => setTimeout(r, 3000))  // aguarda 3s
  }
}

const deployment = await nivo.apps.deploy('app_id', { source, size })
await aguardarDeploy(deployment.id)
```

## Listar deployments

```typescript
const deployments = await nivo.apps.listDeployments('app_id')

for (const dep of deployments) {
  console.log(`${dep.id} — ${dep.status} — ${dep.createdAt}`)
}
```

## Logs de build

```typescript
const logs = await nivo.apps.getDeploymentLogs('app_id', 'deployment_id')
console.log(logs.output)
```

## Caso de uso: CI/CD personalizado

```typescript
import { NivoAPI } from '@nivo-lat/sdk'
import { createZip } from './utils/zip'

const nivo = new NivoAPI(process.env.NIVO_API_KEY!, {
  projectId: process.env.NIVO_PROJECT_ID!
})

async function deploy() {
  console.log('Criando ZIP...')
  const { path, size } = await createZip('./dist')

  console.log('Fazendo deploy...')
  const deployment = await nivo.apps.deploy(process.env.NIVO_APP_ID!, {
    source: createReadStream(path),
    size
  })

  console.log(`Aguardando deploy ${deployment.id}...`)
  await aguardarDeploy(deployment.id)

  console.log('Deploy concluído com sucesso!')
}

deploy().catch(console.error)
```
